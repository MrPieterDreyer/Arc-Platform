    ```ts
    import { useSyncExternalStore, useOptimistic, useCallback } from 'react';
    import type { WCCart } from '../types/cart.js';
    import type { WooClient } from '../http/client.js';
    import { getCart, addItem, updateItem, removeItem, applyCoupon, removeCoupon } from '../store-api/cart.js';
    import type { AddItemPayload, UpdateItemPayload } from '../store-api/cart.js';
    ```

    `CartStore` class (not exported — internal):
    - `private listeners: Set<() => void>` — subscription set.
    - `private cart: WCCart | null = null`.
    - `private loading = false`.
    - `subscribe(listener: () => void): () => void` — adds, returns unsubscribe.
    - `getSnapshot(): CartSnapshot` — returns `{ cart: this.cart, loading: this.loading }`.
    - `getServerSnapshot(): CartSnapshot` — returns `{ cart: null, loading: false }` (SSR-safe).
    - `private notify(): void` — iterates listeners.
    - `async refresh(client: WooClient): Promise<void>` — sets loading, fetches, updates, notifies.
    - `async mutate(client: WooClient, fn: () => Promise<WCCart>): Promise<void>` — calls fn(), updates cart, notifies. Handles ArcRequestError by re-throwing after notify.
    - `setCart(cart: WCCart): void` — sets this.cart, notifies. Used for optimistic updates.

    `const storeMap = new WeakMap<WooClient, CartStore>()` — module-level singleton per WooClient.

    `export function getOrCreateCartStore(client: WooClient): CartStore` — exported for tests.

    Export `CartSnapshot` interface: `{ cart: WCCart | null; loading: boolean }`.

    Export `CartActions` interface:
    ```ts
    export interface CartActions {
      refresh(): Promise<void>;
      addItem(payload: AddItemPayload): Promise<void>;
      updateItem(payload: UpdateItemPayload): Promise<void>;
      removeItem(key: string): Promise<void>;
      applyCoupon(code: string): Promise<void>;
      removeCoupon(code: string): Promise<void>;
    }
    ```

    Export `CartState` = `CartSnapshot & CartActions`.

    `export function useCart(client: WooClient): CartState`:
    - Gets store from `getOrCreateCartStore(client)`.
    - `useSyncExternalStore(store.subscribe, store.getSnapshot, store.getServerSnapshot)`.
    - `useOptimistic(cart, optimisticReducer)` — reducer applies optimistic updates to cart.
    - Returns merged CartSnapshot + CartActions (each action calls `store.mutate(client, () => domainFn(client, ...))`).

    `optimisticReducer(current: WCCart | null, action: OptimisticAction): WCCart | null`:
    - `{ type: 'add', productId: number, quantity: number }` — increment `items_count` by quantity.
    - `{ type: 'remove', key: string }` — decrement `items_count` by removed item's quantity.
    - `{ type: 'update', key: string, quantity: number }` — update item quantity in items array.
    - Returns mutated copy; never mutates in place.

    Do NOT import from `next/*`. File must stay under 200 lines — if longer, split CartStore to `cart-store.ts` and import.
